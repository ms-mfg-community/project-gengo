p6. [x] How do I get an access token for my Azure tenant?
r6. [x]

    # Define the necessary variables for token request
    $tenantId = "your-tenant-id"
    $clientId = "your-client-id"
    $clientSecret = "your-client-secret"
    $scope = "https://management.azure.com/.default"
    $tokenEndpoint = "https://login.microsoftonline.com/$tenantId/oauth2/v2.0/token"

    # Construct the body for the token request
    $body = @{
        client_id     = $clientId
        scope         = $scope
        client_secret = $clientSecret
        grant_type    = "client_credentials"
    }

    # Issue the HTTP POST request to get the access token
    $response = Invoke-RestMethod -Uri $tokenEndpoint -Method Post -ContentType "application/x-www-form-urlencoded" -Body $body

    # Extract the access token
    $accessToken = $response.access_token

    # Define the necessary variables for REST request
    $subscriptionId = "your-subscription-id"
    $resourceGroupName = "your-resource-group-name"
    $devCenterName = "your-devcenter-name"
    $catalogName = "your-catalog-name"
    $apiVersion = "2024-02-01"

    # Construct the URL
    $url = "https://management.azure.com/subscriptions/$subscriptionId/resourceGroups/$resourceGroupName/providers/Microsoft.DevCenter/devcenters/$devCenterName/catalogs/$catalogName/getSyncErrorDetails?api-version=$apiVersion"

    # Define the headers
    $headers = @{
        "Authorization" = "Bearer $accessToken"
        "Content-Type"  = "application/json"
    }

    # Issue the HTTP POST request
    $response = Invoke-RestMethod -Uri $url -Method Post -Headers $headers

    # Output the response
    $response
