# Azure CLI
p01. [] How can I write a azure cloud shell bash script to remove all resource groups that have the string prefix of "rg-alz-mgmt" using a loop ?
r01. [] for rg in $(az group list --query "[?starts_with(name, 'rg-alz-mgmt')].name" -o tsv); do
          az group delete --name $rg --yes --no-wait
        done
