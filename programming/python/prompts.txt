p12. [] @github How many possible combinations are there in the rock paper scissors game ?
r12. [] In the game of rock-paper-scissors, each player has three possible choices: rock, paper, or scissors. Since there are two players, the total number of possible combinations is determined by multiplying the number of choices for each player.
Number of choices for Player 1 = 3 (rock, paper, scissors)
Number of choices for Player 2 = 3 (rock, paper, scissors)
Total possible combinations = 3 * 3 = 9
So, there are 9 possible combinations in a rock-paper-scissors game.

import unittest
from test_game import Game

class TestRockPaperScissors(unittest.TestCase):

    def setUp(self):
        self.game = Game()

    def test_rock_vs_rock(self):
        self.assertEqual(self.game.determine_winner("rock", "rock"), "It's a tie!")

    def test_rock_vs_paper(self):
        self.assertEqual(self.game.determine_winner("rock", "paper"), "Computer wins!")

    def test_rock_vs_scissors(self):
        self.assertEqual(self.game.determine_winner("rock", "scissors"), "You win!")

    def test_paper_vs_rock(self):
        self.assertEqual(self.game.determine_winner("paper", "rock"), "You win!")

    def test_paper_vs_paper(self):
        self.assertEqual(self.game.determine_winner("paper", "paper"), "It's a tie!")

    def test_paper_vs_scissors(self):
        self.assertEqual(self.game.determine_winner("paper", "scissors"), "Computer wins!")

    def test_scissors_vs_rock(self):
        self.assertEqual(self.game.determine_winner("scissors", "rock"), "Computer wins!")
